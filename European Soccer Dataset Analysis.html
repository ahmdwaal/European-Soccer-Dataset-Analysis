'''
Here We Added a New Column "winner_team" Carry the winner team name 
and "result" Carry 'H' if home team win and 'A' if away team win and 'D' if draw
'''
match_with_teams_names_df['winner_team'] = np.where(match_with_teams_names_df['home_team_goal'] == match_with_teams_names_df['away_team_goal'], 'draw', np.where( match_with_teams_names_df['home_team_goal'] >  match_with_teams_names_df['away_team_goal'], match_with_teams_names_df.home_team_name, match_with_teams_names_df.away_team_name))
match_with_teams_names_df['result'] = np.where(match_with_teams_names_df['home_team_goal'] == match_with_teams_names_df['away_team_goal'], 'D', np.where( match_with_teams_names_df['home_team_goal'] >  match_with_teams_names_df['away_team_goal'], 'H', 'A'))
